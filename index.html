        function switchTab(tabId) {

            document.querySelectorAll('.tab-content').forEach(tab => tab.classList.remove('active'));

            document.getElementById(tabId).classList.add('active');

        }

        function drawRandomCards(count) {

            let shuffled = [...tarotDeck].sort(() => 0.5 - Math.random());

            let results = [];

            for(let i=0; i<count; i++) {

                let card = {...shuffled[i]};

                card.isReversed = Math.random() < 0.5; // Automated coin flip

                results.push(card);

            }

            return results;

        }

        function renderCardLayout(cards, labels=[]) {

            return cards.map((card, idx) => `

                <div class="tarot-card">

                    ${labels[idx] ? `<div style="font-size:0.8em; color:var(--gold); font-weight:bold; margin-bottom:5px;">${labels[idx]}</div>` : ''}

                    <div class="card-visual-box ${card.isReversed ? 'reversed' : ''}">✨</div>

                    <div class="card-title">${card.name} ${card.isReversed ? '(Reversed)' : ''}</div>

                    <div class="card-desc"><strong>Energy:</strong> ${card.isReversed ? card.revMeaning : card.meaning}</div>

                </div>

            `).join('');

        }

        function generateSpread() {

            const size = parseInt(document.getElementById('spread-select').value);

            const labels = size === 1 ? ["Clarity"] : ["Past", "Present", "Future"];

            currentActiveReading = drawRandomCards(size);

        function saveCurrentReadingToJournal() {

            const notes = document.getElementById('note-text').value;

            if(!currentActiveReading.length) return;

            let currentLog = JSON.parse(localStorage.getItem('tarot_journal') || '[]');

            const entry = {

                date: new Date().toLocaleString(),

                notes: notes || "No custom annotations added.",

                cards: currentActiveReading

            };

           

            currentLog.unshift(entry);

            localStorage.setItem('tarot_journal', JSON.stringify(currentLog));

            alert("Entry logged securely to device memory.");

            loadJournalEntries();

            switchTab('journal');

        }

        function loadJournalEntries() {

            const log = JSON.parse(localStorage.getItem('tarot_journal') || '[]');

            const container = document.getElementById('journal-entries-list');

            if(!log.length) {

                container.innerHTML = "<p>Your dashboard history log is empty. Save a layout to generate your history tracking graph.</p>";

                return;

            }

            container.innerHTML = log.map(entry => `

                <div class="journal-entry">

                    <div class="journal-date">${entry.date}</div>

                    <div style="font-weight:bold; margin: 5px 0;">Cards Cast: ${entry.cards.map(c => c.name + (c.isReversed ? ' [Rev]':'')).join(', ')}</div>

                    <div style="font-size:0.9em; background:#151124; padding:8px; border-radius:5px; margin-top:5px;"><strong>My Reflections:</strong> ${entry.notes}</div>

                </div>

            `).join('');

        }
